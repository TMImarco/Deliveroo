@section Scripts
{
    <!--Script per la overlay del caricamento-->
    <script>
        document.getElementById("ModificaArticoloFormUpload").addEventListener("submit", function () {
            document.getElementById("ModificaArticoloLoadingOverlay").style.display = "flex";
        });
    </script>
    
    <!--script per adattare la larghezza della <textarea> dells descrizione-->
    <script>
        const ta = document.getElementById("txaModificaDescrizione");
        ta.style.height = ta.scrollHeight + "px";
        ta.addEventListener("input", () => {
            ta.style.height = "auto";
            ta.style.height = ta.scrollHeight + "px";
        });
    </script>

    <!--Script per il pop up-->
    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const modal = document.getElementById('confirm-modal');
            const modalMsg = document.getElementById('modal-msg');
            const btnConfirm = document.getElementById('modal-confirm');
            const btnCancel = document.getElementById('modal-cancel');
            let pendingUrl = null;

            document.querySelectorAll('.confirm-action').forEach(el => {
                el.addEventListener('click', e => {
                    e.preventDefault();
                    pendingUrl = el.getAttribute('href');
                    modalMsg.textContent = el.getAttribute('data-msg');
                    modal.style.display = 'flex';
                });
            });

            btnConfirm.addEventListener('click', () => {
                if (pendingUrl) window.location.href = pendingUrl;
            });

            btnCancel.addEventListener('click', () => {
                modal.style.display = 'none';
                pendingUrl = null;
            });

            modal.addEventListener('click', e => {
                if (e.target === modal) {
                    modal.style.display = 'none';
                    pendingUrl = null;
                }
            });
        });
    </script>
}
